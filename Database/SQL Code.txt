CREATE TABLE PERSONS (
ID IDENTITY PRIMARY KEY,
FNAME NVARCHAR(20) NOT NULL,
LNAME NVARCHAR(20) NOT NULL,
SEX NVARCHAR(5) NOT NULL,
AGE INT NOT NULL,
CHECK (AGE > 0),
SALARY DECIMAL(10,2) NOT NULL
);

CREATE TABLE CARS (
CAR_ID IDENTITY PRIMARY KEY,
BRAND NVARCHAR(20) NOT NULL,
MODEL NVARCHAR(20) NOT NULL,
CAR_YEAR INT NOT NULL,
CHECK (CAR_YEAR > 0),
CHECK (CAR_YEAR <= 2022),
FOR_PERSON INT,
FOREIGN KEY (FOR_PERSON) REFERENCES PERSONS(ID)
);

CREATE TABLE FEES (
FEE_ID IDENTITY PRIMARY KEY,
CONTRIBUTION DECIMAL(10,2) NOT NULL,
CHECK (CONTRIBUTION > 0),
NUMBER_CONTR INT NOT NULL,
CHECK (NUMBER_CONTR > 0),
TOTAL DECIMAL (10,2) NOT NULL,
FOR_PERSON INT,
FOREIGN KEY (FOR_PERSON) REFERENCES PERSONS(ID)
);
